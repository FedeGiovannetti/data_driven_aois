Cluster implementation

peekbank_hdbscan() calculates the spatial clusters with parameters estimated by estimate_dbscan_params().

Here, I run it subject by subject because I was interested in a more individualizaed approach. But it could be done dataset-wise for the import validation purposes (and faster calculations)

clusters_into_gazes() recodes the clusters into “target”, “distractor” and “other” according to the specific trial types. The decision is made by calculatin each cluster’s distance to the target and the distractor fixed aoi centroids.

clustered_data = selected_data %>% 
  group_by(administration_id) %>%
  mutate(clusters = peekbank_hdbscan(pick(everything()))) %>% 
  left_join(trials) 

gazes = clusters_into_gazes(clustered_data) 
df_aois = gazes %>% 
  left_join(trial_types) %>% 
  
  filter(!is.na(x), !is.na(y)) %>% 
  group_by(administration_id, trial_id) %>%
  mutate(n = n()) %>% 
  filter(n >3) %>% 
  select(-n) %>% 
  ungroup() %>% 
  filter(excluded == 0) %>% 
  filter(condition == "familiar-familiar")

summarize_by_subj_by_AOI = df_aois %>%
  filter(excluded == 0) %>% 
  filter(condition == "familiar-familiar") %>% 
  select(administration_id,, x, y, trial_id, t_norm, aoi, new_aoi) %>% 
  pivot_longer(cols = c(aoi, new_aoi), names_to = "AOI", values_to = "position") %>% 
  mutate(AOI = case_when(
    AOI == "aoi" ~ "Fixed",
    AOI == "new_aoi" ~ "Data-driven"
  )) %>% 
  group_by(administration_id, AOI) %>% 
  mutate(result = case_when(
    position == "target" ~ 1,
    position == "distractor" ~ 0,
    TRUE ~ NA_real_
  )) %>%
  group_by(administration_id, AOI, t_norm) %>%
  summarize(
    N = sum(!is.na(result)),
    mean_accuracy = mean(result, na.rm = TRUE),
    .groups = "drop"
  )

#### summarize across subjects ####
n_administrations <- n_distinct(administrations$administration_id)

summarize_across_subj_by_AOI <- summarize_by_subj_by_AOI %>%
  group_by(AOI, t_norm) %>%
  summarize(
    N = sum(!is.na(mean_accuracy)),
    accuracy = mean(mean_accuracy, na.rm = TRUE),
    sd_accuracy = sd(mean_accuracy, na.rm = TRUE),
    se_accuracy = sd_accuracy / sqrt(N),
    .groups = "drop"
  )
